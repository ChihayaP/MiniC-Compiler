define i32 @main() {
    declare i32 %l0
    declare i32 %l1
    declare i32 %t2
    declare i32 %t3
    declare i1 %t4
    declare i32 %t5
    declare i32 %t6
    entry
    %t2 = 1
    %t3 = neg %t2
    %t4 = cmp eq %l0, %t3
    bc %t4, label .L2, label .L1
.L2 :
    %t5 = 1
    exit %t5
.L1 :
    %t6 = 0
    exit %t6
}
